\frametitle{Main functionalities}
\framesubtitle{Reports}
\label{concept:reports-menu}

\begin{block}{Reports}
The \highlight{Reports} menu provides options to save JaBUTi's reports in HTML
format.
\end{block}

\begin{block}{Report types}<2->
\begin{itemize}
	\item<+-> \highlight{Custom Reports} option allows to generate a custom HTML
	report from the current testing project considering different levels of
	granularity.

	\item<+-> \highlight{Summary to HTML} option allows to generate a HTML from
	any tabled style report provided by the JaBUTi graphical interface.
\end{itemize}
\end{block}

\end{frame}
