\framesubtitle{Test suite definition}
\label{example:junit-test-suite}

\begin{lstlisting}
import org.junit.runner.RunWith;
import org.junit.runners.Suite;

@RunWith(Suite.class)
@Suite.SuiteClasses({
    ExampleTestCase.class
})
public class AllTests  {
}
\end{lstlisting}
\end{frame}

	org.junit.runner.JUnitCore
	AllTests
